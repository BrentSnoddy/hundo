                    //starts or controller function
                    //get values from input
                    function getValues(){
                        //get values from the page
                        let startValue = document.getElementById("startValue").value;
                        let endValue = document.getElementById("endValue").value;


                        //we need validate values
                        // parse into integers
                        startValue = parseInt(startValue);
                        endValue = parseInt(endValue);

                        if(Number.isInteger(startValue) && Number.isInteger(endValue)) {
                            //we call generateNumbers function
                            let numbers = generateNumbers(startValue, endValue);
                            //we call displayNumbers function
                            displayNumbers(numbers);
                        } 
                        else{
                            alert("Error! you must enter integers")
                        }

                        
                        
                    }
                    //logic function(s)
                    //generate numbers from start value til the end value
                    function generateNumbers(sValue, eValue){
                        let numbers = [];

                        //get all numbers from start to end
                        for(let index = sValue; index <= eValue; index++){

                            //this will execute in a loop til index = eValue
                            numbers.push(index);
                        }

                        return numbers;

                    }
                    //display or view function
                    //display numbers, even numbers in bold font
                    function displayNumbers(numbers){

                        let templateRows = "";
                        for (let index = 0; index < numbers.length; index++){

                            let className = "even"
                            let number = numbers[index];
                            if (number % 2 == 0){
                                className= "even";

                            } else{
                                className= "odd";

                            }
                            //this does not render correctly with PrismJS. See the source code in Github for full details.//
                            templateRows += `${number}`;
                        }
                        document.getElementById("results").innerHTML = templateRows;

                    }
Hundo Code

Hundo is an app that takes 2 numbers from a user (startingValue and Ending Value) and displays the numbers from the starting number to the ending number. the odd numbers are displayed in a green color and the even numbers are displayed in a purple color. for example: a starting number of 1 and an ending number of 5 would display: 1, 2, 3, 4, 5, with the numbers 2 and the 4 in purple and the numbers 1,3,5 in green. Note: the number 0 is considered an even number in this app.

The Code is structured in 3 functions:

  1. The getValues function:The getValues function takes the input from a user of 2 numbers and first parses the input thereby converting the input from a string to a number. it then checks the input to see if the inout given is an integer. If it is an integer it moves on to the next function otherwise it will give an alert erro message for the user to enter an integer.
  2. generateNumbers function: The generateValues function starts off with a for loop with the starting of the loop being the startingValue variable given by the user and interates a counting sequence till it reaches the endingValue number.it then pushes these values into an array to be used by the next function.
  3. The displayNumbers function:The displayNumbers function takes the array created in the last function and checks to see if the result is divisible by 2 . if it is it will assign the result the class of "Even" thus making it purple. if it s not devisable by 2 it will assign it a class of "Odd" and will make the result a green color in the table. it finally pushes the answer to the table on the page.
